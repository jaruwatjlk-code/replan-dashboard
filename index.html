ลากไฟล์ .xlsx มาวางที่นี่ หรือคลิกเพื่อเลือก
ต้องมี sheet rpall_ และ Target

วิธีตั้งค่า Google Sheets (ทำครั้งเดียว):
1. เปิด Google Sheets → Extensions → Apps Script
2. วางโค้ดนี้แล้ว Save:
function doGet(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var rp=ss.getSheets().find(function(s){return s.getName().toLowerCase().indexOf('rpall_')===0;}); var tg=ss.getSheetByName('Target'); var page=e&&e.parameter&&e.parameter.page?parseInt(e.parameter.page):0; var pageSize=5000; var rpData=[]; var headers=[]; var totalRows=0; if(rp){ var allData=rp.getDataRange().getValues(); headers=allData[0]; totalRows=allData.length-1; var start=1+page*pageSize; var end=Math.min(start+pageSize,allData.length); rpData=allData.slice(page===0?0:start,end); if(page===0)rpData=allData.slice(0,Math.min(1+pageSize,allData.length)); } var d={ rpall:rpData, target:tg?tg.getDataRange().getValues():[], totalRows:totalRows, pageSize:pageSize, page:page }; return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON); }

3. กด Deploy → New deployment → Web App
   — Execute as: Me | Who has access: Anyone
4. คัดลอก URL มาวางด้านล่าง
URL จะถูกจำไว้ — ครั้งหน้าเปิดหน้าเว็บจะโหลดอัตโนมัติ
