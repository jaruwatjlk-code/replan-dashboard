วิธีตั้งค่า Google Sheets (ทำครั้งเดียว):
1. เปิด Google Sheets → Extensions → Apps Script
2. วางโค้ดนี้แล้ว Save:
function doGet(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var rp=ss.getSheets().find(function(s){return s.getName().toLowerCase().indexOf('rpall_')===0;}); var tg=ss.getSheetByName('Target'); var page=e&&e.parameter&&e.parameter.page?parseInt(e.parameter.page):0; var pageSize=5000; var rpData=[]; var headers=[]; var totalRows=0; if(rp){ var allData=rp.getDataRange().getValues(); headers=allData[0]; totalRows=allData.length-1; var start=1+page*pageSize; var end=Math.min(start+pageSize,allData.length); rpData=allData.slice(page===0?0:start,end); if(page===0)rpData=allData.slice(0,Math.min(1+pageSize,allData.length)); } var d={ rpall:rpData, target:tg?tg.getDataRange().getValues():[], totalRows:totalRows, pageSize:pageSize, page:page }; return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON); }

   — Execute as: Me | Who has access: Anyone

function doGet(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var ap=ss.getSheets().find(function(s){return s.getName().toLowerCase().indexOf('apall_')===0;}); var page=e&&e.parameter&&e.parameter.page?parseInt(e.parameter.page):0; var pageSize=5000; var apData=[]; var headers=[]; var totalRows=0; if(ap){ var allData=ap.getDataRange().getValues(); headers=allData[0]; totalRows=allData.length-1; var start=1+page*pageSize; var end=Math.min(start+pageSize,allData.length); var slice=page===0?allData.slice(0,Math.min(1+pageSize,allData.length)):allData.slice(start,end); // แปลง Date object → {year, month} ตรงๆ หลีกเลี่ยงปัญหา locale ทั้งหมด apData=slice.map(function(row){ return row.map(function(cell){ if(cell instanceof Date && !isNaN(cell.getTime())){ return '__DATE__'+cell.getFullYear()+'/'+('0'+(cell.getMonth()+1)).slice(-2); } return cell; }); }); } var d={apall:apData,totalRows:totalRows,pageSize:pageSize,page:page}; return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON); }
